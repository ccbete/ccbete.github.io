Centos 搭建 Tengine Web 服务器高性能访问

发表于 2018-06-11

1、通过yum 命令安装基础环境:

yum -y install wget unzip vim lsof libjpeg libjpeg-devel libpng libpng-devel freetype freetype-devel gd-devel openssl openssl-devel zlib zlib-devel pcre pcre-devel gcc gcc-c++ ncurses-devel perl

2、下载Tengine压缩包: wget http://tengine.taobao.org/download/tengine-2.2.2.tar.gz 

tar xf tengine-2.2.2.tar.gz && cd tengine-2.2.2

3、具体安装步骤:

3.1、编译Tengine 多个特性，合并多个文件为一个请求，去掉HTML空格,及图片处理:

3.2、建立相关文件夹： mkdir -p /var/temp/nginx/client && 

3.3、运行编译命令:  firewall-cmd --zone=public --add-port=80/tcp --permanent && firewall-cmd --zone=public --add-port=443/tcp --permanent && firewall-cmd --reload #开放80、443端口

    groupadd www -- 添加用户www

    useradd -s /sbin/nologin -g www www

    ./configure --user=www --group=www --prefix=/usr/local/nginx --conf-path=/usr/local/nginx/conf/nginx.conf --pid-path=/usr/local/nginx/logs/nginx.pid --lock-path=/var/lock/nginx.lock --error-log-path=/var/log/nginx/error.log --http-log-path=/var/log/nginx/access.log --with-http_gzip_static_module --http-client-body-temp-path=/var/temp/nginx/client --http-proxy-temp-path=/var/temp/nginx/proxy --http-fastcgi-temp-path=/var/temp/nginx/fastcgi --http-uwsgi-temp-path=/var/temp/nginx/uwsgi --http-scgi-temp-path=/var/temp/nginx/scgi --with-http_concat_module --with-http_sub_module=shared --with-http_upstream_check_module --with-http_image_filter_module --with-http_trim_filter_module=shared;

    make && make install

3.3、修改nginx配置文件: cd /usr/local/nginx/conf && mv nginx.conf nginx.conf.bak

    vim nginx.conf

3.4、nginx.conf配置文件如下：

    user  www www;
    worker_processes  auto; #建议为cpu核心数+2

    error_log  /webdata/logs/nginx_error.log  crit;

    pid        /usr/local/nginx/logs/nginx.pid;

    worker_rlimit_nofile 65535;

    events {
        use epoll;
        multi_accept on;
        worker_connections  65535;
    }


    http {
        include       mime.types;
        default_type  application/octet-stream;

        limit_conn_zone $binary_remote_addr zone=addr:1m;
        limit_req_zone $binary_remote_addr zone=one:1m rate=1r/s;

        server_names_hash_bucket_size 128;
        client_header_buffer_size 32k;
        large_client_header_buffers 4 32k;
        client_max_body_size 50m;

        sendfile on; 
        tcp_nopush on; 
        tcp_nodelay on;  
        keepalive_timeout 60;
        types_hash_max_size 2048;

        concat on;
        concat_max_files 20;
        concat_unique on;
        concat_delimiter "\r\n";
        concat_ignore_file_error on;

        gzip on;
        gzip_min_length 1k;
        gzip_disable  "MSIE [1-6]\.";
        gzip_vary on;
        gzip_proxied   any;
        gzip_comp_level 6;
        gzip_buffers     4 16k;
        gzip_http_version 1.1;
        gzip_types   text/plain  text/css  application/json  application/x-javascript  text/xml  application/xml  text/text/javascript;

        # php 的 fastcgi配置
        #
        # fastcgi_connect_timeout 300;
        # fastcgi_send_timeout 300;
        # fastcgi_read_timeout 300;
        # fastcgi_buffer_size 64k;
        # fastcgi_buffers 4 64k;
        # fastcgi_busy_buffers_size 128k;
        # fastcgi_temp_file_write_size 256k;
        #

        proxy_redirect off;
        proxy_set_header Host $host;
        proxy_set_header X-real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        client_body_buffer_size 128k;
        proxy_connect_timeout 90;
        proxy_read_timeout 90;
        proxy_buffer_size 4k;
        proxy_buffers 6 32k;
        proxy_busy_buffers_size 64k;
        proxy_temp_file_write_size 64k;  
        proxy_http_version 1.1;
        proxy_set_header Connection "";

        #后端采用多台tomcat 负载
        #nginx+tomcat+redis实现session共享解决后端多台tomcat负载session问题
        #采用ip 固定访问后端不同的tomcat服务器

       #  upstream tomcat_server{ 
       #     ip_hash; 
       #     server 192.168.10.137;
       #      server 192.168.10.138;
       #     server 192.168.10.139;
       #     server 192.168.10.140;
       # }

        # 配置默认80端口访问重定向到https

        server {
            listen       80;
            server_name 192.168.1.1; 
            rewrite ^/(.*) https://aaa.com/$1 permanent;  
        }

        server {
            listen       80;
            server_name aaa.com; 
            rewrite ^/(.*) https://aaa.com/$1 permanent;  
        }

        server {

            listen 443 default_server ssl;
            ssl on;
            ssl_certificate            /usr/local/nginx/ssl/web.pem;
            ssl_certificate_key        /usr/local/nginx/ssl/web.key;
            ssl_protocols              TLSv1 TLSv1.1 TLSv1.2;
            ssl_session_cache          shared:SSL:10m;
            ssl_session_timeout        10m;
            #优化：Nginx默认使用DHE算法来产生密匙，该加密算法效率很低。可以通过如下命令，删掉了kEDH算法。
            ssl_ciphers ALL:!kEDH!ADH:RC4+RSA:+HIGH:+EXP;    
            ssl_prefer_server_ciphers       on;


            server_name aa.com;
            index index.html index.htm;
            root  /webdata/wwwroot/website/root;

            error_page   404   /webdata/wwwroot/website/root/error.html;

            location ~ \.(html|js|css|gif|jpg|png|bmp|swf)$ {        
                expires 30d; 
            }  

            location /js/ {
                concat on;
                concat_max_files 20;
            }

            location /css/ {
                concat on;
                concat_max_files 20;
            }

           #请求转发给后台tomcat
           # location / {
           #     proxy_http_version 1.1;
           #     proxy_set_header Connection "";
           #     proxy_pass http://tomcat_server;
           # }

            access_log  /webdata/logs/access.log;    
        }        
    }

    至此，nginx 配置结束

3.5:添加nginx开机启动： cd /usr/lib/systemd/system/ && vim nginx.service

    [Unit]    
    Description=nginx    
    After=network.target    

    [Service]    
    Type=forking    
    ExecStart=/usr/local/nginx/sbin/nginx   
    ExecReload=/usr/local/nginx/sbin/nginx -s reload    
    ExecStop=/usr/local/nginx/sbin/nginx -s stop  
    PrivateTmp=true    

    [Install]    
    WantedBy=multi-user.target  

    设置开机启动:

    systemctl enable nginx.service

    systemctl start nginx.service
